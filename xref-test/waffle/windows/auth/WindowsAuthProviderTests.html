67      /**
68       * Test logon user.
69       */
70      @Test
71      public void testLogonUser() {
72          final LMAccess.USER_INFO_1 userInfo = new LMAccess.USER_INFO_1();
73          userInfo.usri1_name = new WString("WaffleTestUser").toString();
74          userInfo.usri1_password = new WString("!WAFFLEP$$Wrd0").toString();
75          userInfo.usri1_priv = LMAccess.USER_PRIV_USER;
76          // ignore test if not able to add user (need to be administrator to do this).
77          Assumptions.assumeTrue(LMErr.NERR_Success == Netapi32.INSTANCE.NetUserAdd(null, 1, userInfo, null));
78          try {
79              final IWindowsAuthProvider prov = new WindowsAuthProviderImpl();
80              final IWindowsIdentity identity = prov.logonUser(userInfo.usri1_name.toString(),
81                      userInfo.usri1_password.toString());
82              Assertions.assertTrue(identity.getFqn().endsWith("\\" + userInfo.usri1_name.toString()));
83              Assertions.assertFalse(identity.isGuest());
84              identity.dispose();
85          } finally {
86              Assertions.assertEquals(LMErr.NERR_Success,
87                      Netapi32.INSTANCE.NetUserDel(null, userInfo.usri1_name.toString()));
88          }
89      }

91      /**
92       * Test impersonate logged on user.
93       */
94      @Test
95      public void testImpersonateLoggedOnUser() {
96          final LMAccess.USER_INFO_1 userInfo = new LMAccess.USER_INFO_1();
97          userInfo.usri1_name = new WString(MockWindowsAccount.TEST_USER_NAME).toString();
98          userInfo.usri1_password = new WString(MockWindowsAccount.TEST_PASSWORD).toString();
99          userInfo.usri1_priv = LMAccess.USER_PRIV_USER;
100         // ignore test if not able to add user (need to be administrator to do this).
101         Assumptions.assumeTrue(LMErr.NERR_Success == Netapi32.INSTANCE.NetUserAdd(null, 1, userInfo, null));
102         try {
103             final IWindowsAuthProvider prov = new WindowsAuthProviderImpl();
104             final IWindowsIdentity identity = prov.logonUser(userInfo.usri1_name.toString(),
105                     userInfo.usri1_password.toString());
106             final IWindowsImpersonationContext ctx = identity.impersonate();
107             Assertions.assertTrue(userInfo.usri1_name.toString().equals(Advapi32Util.getUserName()));
108             ctx.revertToSelf();
109             Assertions.assertFalse(userInfo.usri1_name.toString().equals(Advapi32Util.getUserName()));
110             identity.dispose();
111         } finally {
112             Assertions.assertEquals(LMErr.NERR_Success,
113                     Netapi32.INSTANCE.NetUserDel(null, userInfo.usri1_name.toString()));
114         }

153     /**
154      * Test accept security token.
155      */
156     @Test
157     public void testAcceptSecurityToken() {
158         final String securityPackage = "Negotiate";
159         final String targetName = "localhost";
160         IWindowsCredentialsHandle clientCredentials = null;
161         WindowsSecurityContextImpl clientContext = null;
162         IWindowsSecurityContext serverContext = null;
163         try {
164             // client credentials handle
165             clientCredentials = WindowsCredentialsHandleImpl.getCurrent(securityPackage);
166             clientCredentials.initialize();
167             // initial client security context
168             clientContext = new WindowsSecurityContextImpl();
169             clientContext.setPrincipalName(WindowsAccountImpl.getCurrentUsername());
170             clientContext.setCredentialsHandle(clientCredentials);
171             clientContext.setSecurityPackage(securityPackage);
172             clientContext.initialize(null, null, targetName);
173             // accept on the server
174             final WindowsAuthProviderImpl provider = new WindowsAuthProviderImpl();
175             final String connectionId = "testConnection-" + Thread.currentThread().getId();
176             do {
177                 // accept the token on the server
178                 serverContext = provider.acceptSecurityToken(connectionId, clientContext.getToken(), securityPackage);
179 
180                 if (serverContext != null && serverContext.isContinue()) {
181                     // initialize on the client
182                     final ManagedSecBufferDesc continueToken = new ManagedSecBufferDesc(Sspi.SECBUFFER_TOKEN,
183                             serverContext.getToken());
184                     clientContext.initialize(clientContext.getHandle(), continueToken, targetName);
185                     WindowsAuthProviderTests.LOGGER.info("Token: {}",
186                             Base64.getEncoder().encodeToString(serverContext.getToken()));
187                 }
188 
189             } while (serverContext != null && serverContext.isContinue());
190 
191             if (serverContext != null) {
192                 assertThat(serverContext.getIdentity().getFqn().length()).isGreaterThan(0);
193 
194                 WindowsAuthProviderTests.LOGGER.info(serverContext.getIdentity().getFqn());
195                 for (final IWindowsAccount group : serverContext.getIdentity().getGroups()) {
196                     WindowsAuthProviderTests.LOGGER.info(" {}", group.getFqn());
197                 }
198             }
199         } finally {
200             if (serverContext != null) {
201                 serverContext.dispose();
202             }
203             if (clientContext != null) {
204                 clientContext.dispose();
205             }
206             if (clientCredentials != null) {
207                 clientCredentials.dispose();
208             }
209         }
210     }

212     /**
213      * Test security contexts expire.
214      *
215      * @throws InterruptedException
216      *             the interrupted exception
217      */
218     @Test
219     public void testSecurityContextsExpire() throws InterruptedException {
220         final String securityPackage = "Negotiate";
221         IWindowsCredentialsHandle clientCredentials = null;
222         WindowsSecurityContextImpl clientContext = null;
223         IWindowsSecurityContext serverContext = null;
224         try {
225             // client credentials handle
226             clientCredentials = WindowsCredentialsHandleImpl.getCurrent(securityPackage);
227             clientCredentials.initialize();
228             // initial client security context
229             clientContext = new WindowsSecurityContextImpl();
230             clientContext.setPrincipalName(WindowsAccountImpl.getCurrentUsername());
231             clientContext.setCredentialsHandle(clientCredentials);
232             clientContext.setSecurityPackage(securityPackage);
233             clientContext.initialize(null, null, WindowsAccountImpl.getCurrentUsername());
234             // accept on the server
235             final WindowsAuthProviderImpl provider = new WindowsAuthProviderImpl(1);
236             final int max = 100;
237             for (int i = 0; i < max; i++) {
238                 Thread.sleep(25);
239                 final String connectionId = "testConnection_" + i;
240                 serverContext = provider.acceptSecurityToken(connectionId, clientContext.getToken(), securityPackage);
241                 assertThat(provider.getContinueContextsSize()).isGreaterThan(0);
242             }
243             WindowsAuthProviderTests.LOGGER.info("Cached security contexts: {}",
244                     Integer.valueOf(provider.getContinueContextsSize()));
245             Assertions.assertFalse(max == provider.getContinueContextsSize());
246         } finally {
247             if (serverContext != null) {
248                 serverContext.dispose();
249             }
250             if (clientContext != null) {
251                 clientContext.dispose();
252             }
253             if (clientCredentials != null) {
254                 clientCredentials.dispose();
255             }
256         }
257     }
