12  package waffle.apache;

14  import java.security.Principal;
15  
16  import org.apache.catalina.realm.RealmBase;
17  
18  /**
19   * A rudimentary Windows realm.
20   *
21   * @author dblock[at]dblock[dot]org
22   */
23  public class WindowsRealm extends RealmBase {
24  
25      /** The Constant NAME. */
26      protected static final String NAME = "waffle.apache.WindowsRealm/1.0";
27  
28      /**
29       * Gets the name.
30       *
31       * 'waffle.apache.WindowsRealm/1.0' will no longer be logged. We don't internally use this so we must go with
32       * standard java way that tomcat has accepted. This means, going to tomcat 9.0.0.M15+ will result simply in
33       * 'WaffleRealm' or better stated the actual simple class name. Simple class name strips off the package name with
34       * is what we were applying along with version 1.0 which is inaccurate anyways considering we are on version 1.8.1+
35       * at this point.
36       *
37       * @return a short name for this Realm implementation, for use in log messages.
38       * 
39       * @deprecated This will be removed in Tomcat 9 onwards. Use {@link Class#getSimpleName()} instead.
40       */
41      @Deprecated
42      protected String getName() {
43          return WindowsRealm.NAME;
44      }
45  
46      @Override
47      protected String getPassword(final String value) {
48          return null;
49      }
50  
51      @Override
52      protected Principal getPrincipal(final String value) {
53          return null;
54      }
55  
56  }
