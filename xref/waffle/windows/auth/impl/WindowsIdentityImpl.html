20  import java.util.ArrayList;

22  
23  import waffle.windows.auth.IWindowsAccount;
24  import waffle.windows.auth.IWindowsIdentity;

27  /**
28   * Windows Identity.
29   *
30   * @author dblock[at]dblock[dot]org
31   */
32  public class WindowsIdentityImpl implements IWindowsIdentity {
33  
34      /** The windows identity. */
35      private final HANDLE windowsIdentity;
36  
37      /** The user groups. */
38      private Account[] userGroups;
39  
40      /** The windows account. */
41      private Account windowsAccount;
42  
43      /**
44       * Instantiates a new windows identity impl.
45       *
46       * @param newWindowsIdentity
47       *            the new windows identity
48       */
49      public WindowsIdentityImpl(final HANDLE newWindowsIdentity) {
50          this.windowsIdentity = newWindowsIdentity;
51      }
52  
53      /**
54       * Gets the windows account.
55       *
56       * @return the windows account
57       */
58      private Account getWindowsAccount() {
59          if (this.windowsAccount == null) {
60              this.windowsAccount = Advapi32Util.getTokenAccount(this.windowsIdentity);
61          }
62          return this.windowsAccount;
63      }
64  
65      /**
66       * Gets the user groups.
67       *
68       * @return the user groups
69       */
70      private Account[] getUserGroups() {
71          if (this.userGroups == null) {
72              this.userGroups = Advapi32Util.getTokenGroups(this.windowsIdentity);
73          }
74          return this.userGroups.clone();
75      }
76  
77      @Override
78      public String getFqn() {
79          return this.getWindowsAccount().fqn;
80      }
81  
82      @Override
83      public IWindowsAccount[] getGroups() {
84  
85          final Account[] groups = this.getUserGroups();
86  
87          final List<IWindowsAccount> result = new ArrayList<>(groups.length);
88          for (final Account userGroup : groups) {
89              final WindowsAccountImplpl.html#WindowsAccountImpl">WindowsAccountImpl account = new WindowsAccountImpl(userGroup);
90              result.add(account);
91          }
92  
93          return result.toArray(new IWindowsAccount[0]);
94      }
95  
96      @Override
97      public byte[] getSid() {
98          return this.getWindowsAccount().sid;
99      }
100 
101     @Override
102     public String getSidString() {
103         return this.getWindowsAccount().sidString;
104     }
105 
106     @Override
107     public void dispose() {
108         if (this.windowsIdentity != null) {
109             Kernel32.INSTANCE.CloseHandle(this.windowsIdentity);
110         }
111     }
112 
113     @Override
114     public IWindowsImpersonationContext impersonate() {
115         return new WindowsIdentityImpersonationContextImpl(this.windowsIdentity);
116     }
117 
118     @Override
119     public boolean isGuest() {
120         for (final Account userGroup : this.getUserGroups()) {
121             if (Advapi32Util.isWellKnownSid(userGroup.sid, WELL_KNOWN_SID_TYPE.WinBuiltinGuestsSid)) {
122                 return true;
123             }
124             if (Advapi32Util.isWellKnownSid(userGroup.sid, WELL_KNOWN_SID_TYPE.WinAccountDomainGuestsSid)) {
125                 return true;
126             }
127             if (Advapi32Util.isWellKnownSid(userGroup.sid, WELL_KNOWN_SID_TYPE.WinAccountGuestSid)) {
128                 return true;
129             }
130         }
131         return Advapi32Util.isWellKnownSid(this.getSid(), WELL_KNOWN_SID_TYPE.WinAnonymousSid);
132     }
133 }
