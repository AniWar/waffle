12  package waffle.shiro;
13  
14  import java.io.Serializable;
15  import java.util.Collections;
16  import java.util.HashSet;
17  import java.util.Set;
18  
19  import waffle.windows.auth.IWindowsAccount;
20  import waffle.windows.auth.IWindowsIdentity;
21  
22  /**
23   * The Class WaffleFqnPrincipal.
24   */
25  public class WaffleFqnPrincipal implements Serializable {
26  
27      /** The Constant serialVersionUID. */
28      private static final long serialVersionUID = 1;
29  
30      /** The fqn. */
31      private final String fqn;
32  
33      /** The group fqns. */
34      private final Set<String> groupFqns = new HashSet<>();
35  
36      /**
37       * Instantiates a new waffle fqn principal.
38       *
39       * @param identity
40       *            the identity
41       */
42      WaffleFqnPrincipal(final IWindowsIdentity identity) {
43          this.fqn = identity.getFqn();
44          for (final IWindowsAccount group : identity.getGroups()) {
45              this.groupFqns.add(group.getFqn());
46          }
47      }
48  
49      /**
50       * Gets the fqn.
51       *
52       * @return the fully qualified name of the user
53       */
54      public String getFqn() {
55          return this.fqn;
56      }
57  
58      /**
59       * Gets the group fqns.
60       *
61       * @return the fully qualified names of all groups that the use belongs to
62       */
63      public Set<String> getGroupFqns() {
64          return Collections.unmodifiableSet(this.groupFqns);
65      }
66  
67      @Override
68      public boolean equals(final Object obj) {
69          if (obj instanceof WaffleFqnPrincipal) {
70              return this.fqn.equals(((WaffleFqnPrincipal) obj).fqn);
71          }
72          return false;
73      }
74  
75      @Override
76      public int hashCode() {
77          return this.fqn.hashCode();
78      }
79  
80      @Override
81      public String toString() {
82          final StringBuilder stringBuilder = new StringBuilder();
83          stringBuilder.append("{");
84          stringBuilder.append(this.getClass().getSimpleName());
85          stringBuilder.append(":");
86          stringBuilder.append(this.fqn);
87          stringBuilder.append("}");
88          return stringBuilder.toString();
89      }
90  }
