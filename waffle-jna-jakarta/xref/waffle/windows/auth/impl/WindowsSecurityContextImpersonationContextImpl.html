12  package waffle.windows.auth.impl;
13  
14  import com.sun.jna.platform.win32.Secur32;
15  import com.sun.jna.platform.win32.Sspi.CtxtHandle;
16  import com.sun.jna.platform.win32.Win32Exception;
17  import com.sun.jna.platform.win32.WinError;
18  
19  import waffle.windows.auth.IWindowsImpersonationContext;
20  
21  /**
22   * The Class WindowsSecurityContextImpersonationContextImpl.
23   *
24   * @author dblock[at]dblock[dot]org
25   */
26  public class WindowsSecurityContextImpersonationContextImpl implements IWindowsImpersonationContext {
27  
28      /** The ctx. */
29      private final CtxtHandle ctx;
30  
31      /**
32       * Instantiates a new windows security context impersonation context impl.
33       *
34       * @param newCtx
35       *            the new ctx
36       */
37      public WindowsSecurityContextImpersonationContextImpl(final CtxtHandle newCtx) {
38          final int rc = Secur32.INSTANCE.ImpersonateSecurityContext(newCtx);
39          if (rc != WinError.SEC_E_OK) {
40              throw new Win32Exception(rc);
41          }
42  
43          this.ctx = newCtx;
44      }
45  
46      @Override
47      public void revertToSelf() {
48          final int rc = Secur32.INSTANCE.RevertSecurityContext(this.ctx);
49          if (rc != WinError.SEC_E_OK) {
50              throw new Win32Exception(rc);
51          }
52      }
53  }
