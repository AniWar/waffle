14  import org.junit.jupiter.api.Assertions;

17  /**

22  public class NtlmMessageTests {
23  
24      /**
25       * Test is ntlm message.
26       */
27      @Test
28      public void testIsNtlmMessage() {
29          Assertions.assertFalse(NtlmMessage.isNtlmMessage(null));
30          final byte[] ntlmSignature = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00 };
31          Assertions.assertTrue(NtlmMessage.isNtlmMessage(ntlmSignature));
32          final byte[] shortMessage = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50 };
33          Assertions.assertFalse(NtlmMessage.isNtlmMessage(shortMessage));
34          final byte[] longMessage = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x00 };
35          Assertions.assertTrue(NtlmMessage.isNtlmMessage(longMessage));
36          final byte[] badMessage = { 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00 };
37          Assertions.assertFalse(NtlmMessage.isNtlmMessage(badMessage));
38      }
39  
40      /**
41       * Test get ntlm message type.
42       */
43      @Test
44      public void testGetNtlmMessageType() {
45          final byte[] ntlmMessageType1 = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x01, 0x00, 0x00, 0x00, 0x02,
46                  0x02, 0x00, 0x00 };
47          Assertions.assertEquals(1, NtlmMessage.getMessageType(ntlmMessageType1));
48          final byte[] ntlmMessageType2 = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x02, 0x00, 0x00, 0x00, 0x00,
49                  0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x02, 0x02, 0x00, 0x00, 0x01, 0x23, 0x45, 0x67 };
50          Assertions.assertEquals(2, NtlmMessage.getMessageType(ntlmMessageType2));
51          final byte[] ntlmMessageType3 = { 0x4e, 0x54, 0x4c, 0x4d, 0x53, 0x53, 0x50, 0x00, 0x03, 0x00, 0x00, 0x00, 0x00,
52                  0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x02, 0x02, 0x00, 0x00, 0x01, 0x23, 0x45, 0x67 };
53          Assertions.assertEquals(3, NtlmMessage.getMessageType(ntlmMessageType3));
54      }
