12  package waffle.windows.auth.impl;
13  
14  import com.sun.jna.platform.win32.Advapi32;
15  import com.sun.jna.platform.win32.Kernel32;
16  import com.sun.jna.platform.win32.Win32Exception;
17  import com.sun.jna.platform.win32.WinNT.HANDLE;
18  
19  import waffle.windows.auth.IWindowsImpersonationContext;
20  
21  /**
22   * The Class WindowsIdentityImpersonationContextImpl.
23   *
24   * @author dblock[at]dblock[dot]org
25   */
26  public class WindowsIdentityImpersonationContextImpl implements IWindowsImpersonationContext {
27  
28      /**
29       * Impersonate a logged on user.
30       *
31       * @param windowsIdentity
32       *            Windows identity obtained via LogonUser.
33       */
34      public WindowsIdentityImpersonationContextImpl(final HANDLE windowsIdentity) {
35          if (!Advapi32.INSTANCE.ImpersonateLoggedOnUser(windowsIdentity)) {
36              throw new Win32Exception(Kernel32.INSTANCE.GetLastError());
37          }
38      }
39  
40      @Override
41      public void revertToSelf() {
42          Advapi32.INSTANCE.RevertToSelf();
43      }
44  }
